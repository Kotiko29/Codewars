function dontGiveMeFive(start, end){ 
  let result = 0; 
  for (var i = start; i <= end; i++) { 
    let j = i + ""; 
    let reg = j.search(/5/g); 
    if (reg == -1) { 
      result++ 
    } 
  } 
  return result 
}

function dontGiveMeFive(start, end) {
  let count = 0
  for (let i = start; i <= end; i++) {
    if (!/5/.test(i)) {
      count++
    }
  }
  return count
